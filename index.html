        let currentQuestion = 0;

        let answers = new Array(questions.length).fill(null);


        function startQuiz() {

            document.getElementById('welcomePage').style.display = 'none';

            document.getElementById('quizPage').style.display = 'block';

            renderQuestion();

        }


        function renderQuestion() {

            const q = questions[currentQuestion];

            const card = document.getElementById('questionCard');

            let html = `<div class="question-number">QUESTION ${currentQuestion + 1}</div>

                <div class="question-text">${q.text}</div>

                <div class="options">`;

            q.options.forEach((opt, idx) => {

                const letter = String.fromCharCode(65 + idx);

                const selected = answers[currentQuestion] === idx ? 'selected' : '';

                html += `<div class="option ${selected}" onclick="selectOption(${idx})">

                    <div class="option-content">

                        <span class="option-letter">${letter}</span>

                        <span>${opt.text}</span>

                    </div>

                </div>`;

            });

            html += '</div>';

            card.innerHTML = html;

            const progress = ((currentQuestion + 1) / questions.length) * 100;

            document.getElementById('progressFill').style.width = progress + '%';

            document.getElementById('progressText').textContent = `第 ${currentQuestion + 1} / ${questions.length} 题`;

            document.getElementById('prevBtn').disabled = currentQuestion === 0;

            const nextBtn = document.getElementById('nextBtn');

            if (answers[currentQuestion] !== null) {

                nextBtn.style.display = 'block';

                nextBtn.textContent = currentQuestion === questions.length - 1 ? '查看报告' : '下一题';

            } else {

                nextBtn.style.display = 'none';

            }

        }


        function selectOption(idx) {

            answers[currentQuestion] = idx;

            const options = document.querySelectorAll('.option');

            options.forEach((opt, i) => {

                if (i === idx) opt.classList.add('selected');

                else opt.classList.remove('selected');

            });

            const nextBtn = document.getElementById('nextBtn');

            nextBtn.style.display = 'block';

            if (currentQuestion === questions.length - 1) nextBtn.textContent = '查看报告';

        }


        function nextQuestion() {

            if (currentQuestion < questions.length - 1) {

                currentQuestion++;

                renderQuestion();

            } else {

                generateReport();

            }

        }


        function prevQuestion() {

            if (currentQuestion > 0) {

                currentQuestion--;

                renderQuestion();

            }

        }

        function calculateScores() {

            const scores = {

                D1: { behavior: [], cognitive: [] },

                D2: { behavior: [], cognitive: [] },

                D3: { behavior: [], cognitive: [] },

                D4: { behavior: [], cognitive: [] },

                D5: { behavior: [], cognitive: [] },

                D6: { behavior: [], cognitive: [] },

                error_book: { organize: [], use: [] },

                validity: { check1: null, check2: null }

            };

            

            questions.forEach((q, idx) => {

                const answerIdx = answers[idx];

                if (answerIdx === null) return;

                const score = q.options[answerIdx].score;

                if (q.dimension === 'validity') {

                    scores.validity[q.track] = answerIdx;

                } else if (q.dimension === 'error_book') {

                    scores.error_book[q.track].push(score);

                } else {

                    scores[q.dimension][q.track].push(score);

                }

            });

            

            const dimensions = {};

            ['D1', 'D2', 'D3', 'D4', 'D5', 'D6'].forEach(d => {

                const behaviorAvg = scores[d].behavior.length > 0 

                    ? scores[d].behavior.reduce((a, b) => a + b, 0) / scores[d].behavior.length 

                    : 0;

                const cognitiveAvg = scores[d].cognitive.length > 0 

                    ? scores[d].cognitive.reduce((a, b) => a + b, 0) / scores[d].cognitive.length 

                    : 0;

                dimensions[d] = {

                    behavior: behaviorAvg,

                    cognitive: cognitiveAvg,

                    total: behaviorAvg * 0.7 + cognitiveAvg * 0.3

                };

            });

            

            const organizeAvg = scores.error_book.organize.length > 0

                ? scores.error_book.organize.reduce((a, b) => a + b, 0) / scores.error_book.organize.length

                : 0;

            const useAvg = scores.error_book.use.length > 0

                ? scores.error_book.use.reduce((a, b) => a + b, 0) / scores.error_book.use.length

                : 0;

            dimensions.error_book = {

                organize: organizeAvg,

                use: useAvg,

                total: (organizeAvg + useAvg) / 2

            };

            

            const validityDiff = Math.abs(scores.validity.check1 - scores.validity.check2);

            const validityPassed = validityDiff <= 1;

            

            return { dimensions, validityPassed, validityDiff };

        }
